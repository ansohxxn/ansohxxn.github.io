int solution(vector<vector<int>> land)

    int answer = 0;

    vector<int> fourColRecord(4);

    int selectedIndex = 0;
    for (int i = 0; i < fourColRecord.size(); i++) {

                fourColRecord[i] = land[0][i];
                selectedIndex = i;

                int max = 0;
                int index = 0;
                for (int k = 0; k < 4; k++) {
                    if (k != selectedIndex && max <= land[j][k]) {
                        max = land[j][k];
                        index = k;
                    }
                }
                selectedIndex = index;
                fourColRecord[i] += max;
            }
        }
    }
    
    answer = *max_element(fourColRecord.begin(), fourColRecord.end());
    
    return answer;
}

            int max = 0;

                if (k == j) continue;
                if (max < dp[i - 1][k])
                    max = dp[i - 1][k]; 
            }   
            dp[i][j] = land[i][j] + max;
        }
    }
    
    answer = *max_element(dp.back().begin(), dp.back().end());
    
    return answer;
}
