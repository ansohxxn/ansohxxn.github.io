    public float magnitude { get { return Mathf.Sqrt(x * x + y * y + z * z); } }

    {
        return new MyVector(a.x - b.x, a.y - b.y, a.z - b.z);
    }

    public static MyVector operator * (MyVector a, MyVector b)
    {
        return new MyVector(a.x * b.x, a.y * b.y, a.z * b.z);
    }
}
